USAGE                                                        *hurl.nvim-usage*

>lua
      {
        "jellydn/hurl.nvim",
        ft = "hurl",
        cmd = { "HurlRun" },
        opts = {
          debug = true, -- default is false
        },
        keys = {
          -- Run API request
          { "<leader>ra", "<cmd>HurlRun<CR>", desc = "Run API requests" },
          -- Run API request in visual mode
          { "<leader>cr", ":HurlRun<CR>", desc = "Run API request", mode = "v" },
        },
      }
    }
